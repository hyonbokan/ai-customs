# 1. Stop & remove any existing container
sudo docker rm -f customs-ai-container || true

# 2. (Re)build your image from the current directory
sudo docker build -t customs-ai-backend -f backend/Dockerfile .
sudo docker build --no-cache -t customs-ai-backend -f backend/Dockerfile .

# 3. Launch a new container, mapping host 8000→container 8000
sudo docker run \
  --name customs-ai-container \
  -p 8000:8000 \
  customs-ai-backend

sudo docker run \
  --name customs-ai-container \
  -p 8000:8000 \
  --add-host=host.docker.internal:host-gateway \
  -e TGI_BASE_URL=http://host.docker.internal:8080/v1/ \
  customs-ai-backend
